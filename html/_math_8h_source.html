719 static inline void mat4x4_rotate(mat4x4 R, mat4x4 M, float x, float y, float z, float angle)
720 {
721  float s = sinf(angle);
722  float c = cosf(angle);
723 
724  vec3 u = {x, y, z};
725 
726  if(vec3_len(u) > 1e-4) {
727  vec3_norm(u, u);
728  mat4x4 T;
729  mat4x4_from_vec3_mul_outer(T, u, u);
730  mat4x4 S = {
731  { 0, u[2], -u[1], 0},
732  {-u[2], 0, u[0], 0},
733  { u[1], -u[0], 0, 0},
734  { 0, 0, 0, 0}
735  };
736 
737  mat4x4_scale(S, S, s);
738  mat4x4 C;
739 
740  mat4x4_identity(C);
741  mat4x4_sub(C, C, T);
742  mat4x4_scale(C, C, c);
743  mat4x4_add(T, T, C);
744  mat4x4_add(T, T, S);
745 
746  T[3][3] = 1.;
747  mat4x4_mul(R, M, T);
748  } else {
749  mat4x4_dup(R, M);
750  }
751 }

830 static inline void mat4x4_invert(mat4x4 T, mat4x4 M)
831 {
832  float s[6];
833  float c[6];
834 
835  s[0] = M[0][0]*M[1][1] - M[1][0]*M[0][1];
836  s[1] = M[0][0]*M[1][2] - M[1][0]*M[0][2];
837  s[2] = M[0][0]*M[1][3] - M[1][0]*M[0][3];
838  s[3] = M[0][1]*M[1][2] - M[1][1]*M[0][2];
839  s[4] = M[0][1]*M[1][3] - M[1][1]*M[0][3];
840  s[5] = M[0][2]*M[1][3] - M[1][2]*M[0][3];
841 
842  c[0] = M[2][0]*M[3][1] - M[3][0]*M[2][1];
843  c[1] = M[2][0]*M[3][2] - M[3][0]*M[2][2];
844  c[2] = M[2][0]*M[3][3] - M[3][0]*M[2][3];
845  c[3] = M[2][1]*M[3][2] - M[3][1]*M[2][2];
846  c[4] = M[2][1]*M[3][3] - M[3][1]*M[2][3];
847  c[5] = M[2][2]*M[3][3] - M[3][2]*M[2][3];
848 
849  /* Assumes it is invertible */
850  float idet = 1.0f/( s[0]*c[5]-s[1]*c[4]+s[2]*c[3]+s[3]*c[2]-s[4]*c[1]+s[5]*c[0] );
851 
852  T[0][0] = ( M[1][1] * c[5] - M[1][2] * c[4] + M[1][3] * c[3]) * idet;
853  T[0][1] = (-M[0][1] * c[5] + M[0][2] * c[4] - M[0][3] * c[3]) * idet;
854  T[0][2] = ( M[3][1] * s[5] - M[3][2] * s[4] + M[3][3] * s[3]) * idet;
855  T[0][3] = (-M[2][1] * s[5] + M[2][2] * s[4] - M[2][3] * s[3]) * idet;
856 
857  T[1][0] = (-M[1][0] * c[5] + M[1][2] * c[2] - M[1][3] * c[1]) * idet;
858  T[1][1] = ( M[0][0] * c[5] - M[0][2] * c[2] + M[0][3] * c[1]) * idet;
859  T[1][2] = (-M[3][0] * s[5] + M[3][2] * s[2] - M[3][3] * s[1]) * idet;
860  T[1][3] = ( M[2][0] * s[5] - M[2][2] * s[2] + M[2][3] * s[1]) * idet;
861 
862  T[2][0] = ( M[1][0] * c[4] - M[1][1] * c[2] + M[1][3] * c[0]) * idet;
863  T[2][1] = (-M[0][0] * c[4] + M[0][1] * c[2] - M[0][3] * c[0]) * idet;
864  T[2][2] = ( M[3][0] * s[4] - M[3][1] * s[2] + M[3][3] * s[0]) * idet;
865  T[2][3] = (-M[2][0] * s[4] + M[2][1] * s[2] - M[2][3] * s[0]) * idet;
866 
867  T[3][0] = (-M[1][0] * c[3] + M[1][1] * c[1] - M[1][2] * c[0]) * idet;
868  T[3][1] = ( M[0][0] * c[3] - M[0][1] * c[1] + M[0][2] * c[0]) * idet;
869  T[3][2] = (-M[3][0] * s[3] + M[3][1] * s[1] - M[3][2] * s[0]) * idet;
870  T[3][3] = ( M[2][0] * s[3] - M[2][1] * s[1] + M[2][2] * s[0]) * idet;
871 }

905 static inline void mat4x4_frustum(mat4x4 M, float l, float r, float b, float t, float n, float f)
906 {
907  M[0][0] = 2.f*n/(r-l);
908  M[0][1] = M[0][2] = M[0][3] = 0.f;
909 
910  M[1][1] = 2.*n/(t-b);
911  M[1][0] = M[1][2] = M[1][3] = 0.f;
912 
913  M[2][0] = (r+l)/(r-l);
914  M[2][1] = (t+b)/(t-b);
915  M[2][2] = -(f+n)/(f-n);
916  M[2][3] = -1.f;
917 
918  M[3][2] = -2.f*(f*n)/(f-n);
919  M[3][0] = M[3][1] = M[3][3] = 0.f;
920 }

922 static inline void mat4x4_ortho(mat4x4 M, float l, float r, float b, float t, float n, float f)
923 {

928  M[1][0] = M[1][2] = M[1][3] = 0.f;
929 
930  M[2][2] = -2.f/(f-n);
931  M[2][0] = M[2][1] = M[2][3] = 0.f;
932 
933  M[3][0] = -(r+l)/(r-l);
934  M[3][1] = -(t+b)/(t-b);
935  M[3][2] = -(f+n)/(f-n);
936  M[3][3] = 1.f;
