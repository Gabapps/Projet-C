726 static inline void mat4x4_rotate(mat4x4 R, mat4x4 M, float x, float y, float z, float angle)
727 {
728  float s = sinf(angle);
729  float c = cosf(angle);
730 
731  vec3 u = {x, y, z};
732 
733  if(vec3_len(u) > 1e-4) {
734  vec3_norm(u, u);
735  mat4x4 T;
736  mat4x4_from_vec3_mul_outer(T, u, u);
737  mat4x4 S = {
738  { 0, u[2], -u[1], 0},
739  {-u[2], 0, u[0], 0},
740  { u[1], -u[0], 0, 0},
741  { 0, 0, 0, 0}
742  };
743 
744  mat4x4_scale(S, S, s);
745  mat4x4 C;
746 
747  mat4x4_identity(C);
748  mat4x4_sub(C, C, T);
749  mat4x4_scale(C, C, c);
750  mat4x4_add(T, T, C);
751  mat4x4_add(T, T, S);
752 
753  T[3][3] = 1.;
754  mat4x4_mul(R, M, T);
755  } else {
756  mat4x4_dup(R, M);
757  }
758 }

837 static inline void mat4x4_invert(mat4x4 T, mat4x4 M)
838 {
839  float s[6];
840  float c[6];
841 
842  s[0] = M[0][0]*M[1][1] - M[1][0]*M[0][1];
843  s[1] = M[0][0]*M[1][2] - M[1][0]*M[0][2];
844  s[2] = M[0][0]*M[1][3] - M[1][0]*M[0][3];
845  s[3] = M[0][1]*M[1][2] - M[1][1]*M[0][2];
846  s[4] = M[0][1]*M[1][3] - M[1][1]*M[0][3];
847  s[5] = M[0][2]*M[1][3] - M[1][2]*M[0][3];
848 
849  c[0] = M[2][0]*M[3][1] - M[3][0]*M[2][1];
850  c[1] = M[2][0]*M[3][2] - M[3][0]*M[2][2];
851  c[2] = M[2][0]*M[3][3] - M[3][0]*M[2][3];
852  c[3] = M[2][1]*M[3][2] - M[3][1]*M[2][2];
853  c[4] = M[2][1]*M[3][3] - M[3][1]*M[2][3];
854  c[5] = M[2][2]*M[3][3] - M[3][2]*M[2][3];
855 
856  /* Assumes it is invertible */
857  float idet = 1.0f/( s[0]*c[5]-s[1]*c[4]+s[2]*c[3]+s[3]*c[2]-s[4]*c[1]+s[5]*c[0] );
858 
859  T[0][0] = ( M[1][1] * c[5] - M[1][2] * c[4] + M[1][3] * c[3]) * idet;
860  T[0][1] = (-M[0][1] * c[5] + M[0][2] * c[4] - M[0][3] * c[3]) * idet;
861  T[0][2] = ( M[3][1] * s[5] - M[3][2] * s[4] + M[3][3] * s[3]) * idet;
862  T[0][3] = (-M[2][1] * s[5] + M[2][2] * s[4] - M[2][3] * s[3]) * idet;
863 
864  T[1][0] = (-M[1][0] * c[5] + M[1][2] * c[2] - M[1][3] * c[1]) * idet;
865  T[1][1] = ( M[0][0] * c[5] - M[0][2] * c[2] + M[0][3] * c[1]) * idet;
866  T[1][2] = (-M[3][0] * s[5] + M[3][2] * s[2] - M[3][3] * s[1]) * idet;
867  T[1][3] = ( M[2][0] * s[5] - M[2][2] * s[2] + M[2][3] * s[1]) * idet;
868 
869  T[2][0] = ( M[1][0] * c[4] - M[1][1] * c[2] + M[1][3] * c[0]) * idet;
870  T[2][1] = (-M[0][0] * c[4] + M[0][1] * c[2] - M[0][3] * c[0]) * idet;
871  T[2][2] = ( M[3][0] * s[4] - M[3][1] * s[2] + M[3][3] * s[0]) * idet;
872  T[2][3] = (-M[2][0] * s[4] + M[2][1] * s[2] - M[2][3] * s[0]) * idet;
873 
874  T[3][0] = (-M[1][0] * c[3] + M[1][1] * c[1] - M[1][2] * c[0]) * idet;
875  T[3][1] = ( M[0][0] * c[3] - M[0][1] * c[1] + M[0][2] * c[0]) * idet;
876  T[3][2] = (-M[3][0] * s[3] + M[3][1] * s[1] - M[3][2] * s[0]) * idet;
877  T[3][3] = ( M[2][0] * s[3] - M[2][1] * s[1] + M[2][2] * s[0]) * idet;
878 }
